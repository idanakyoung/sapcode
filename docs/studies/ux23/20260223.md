# 1. ZDEMO_03_G01 : 테이블 컬럼 클릭 시 데이터 바인딩 + CREATE & UPDATE & DELETE [ZGW_NAVI_G01]

## 1.1 프로젝트 개요

이 예제는 SAPUI5에서 다음 4가지를 한 화면에 합친 실습이다.

- 검색 조건으로 `ConnectSet` 조회
- 테이블 행 선택 시 상세 영역 바인딩
- 상세 영역에서 Create / Update / Delete 수행
- `mode`라는 JSONModel로 입력 가능/불가능 상태 제어

즉, 화면 구조는 아래와 같다.

```text
검색 조건 입력
   ↓
ConnectSet 목록 조회
   ↓
테이블 행 선택
   ↓
Detail Panel에 선택 데이터 바인딩
   ↓
Insert / Update / Delete 실행
```

## 1.2 전체 코드

## Main Controller

```javascript
sap.ui.define([
    "sap/ui/core/mvc/Controller"
], (Controller) => {
    "use strict";

    return Controller.extend("code.zdemo03g01.controller.Main", {
        onInit() {
            // Detail Info 부분의 값들이 output만 가능하게
            let oData = {
                idEdit: false
            };

            let oModel = new sap.ui.model.json.JSONModel(oData);
            this.getView().setModel(oModel,"mode");
        },
        onSearch: function () {
            let aFilters = [];

            let cityFrom = this.byId("idFromInput").getValue();
            if (cityFrom) {
                aFilters.push(new sap.ui.model.Filter("Cityfrom", sap.ui.model.FilterOperator.Contains, cityFrom))
            };
            
            let cityTo = this.byId("idToInput").getValue();
            if (cityTo) {
                aFilters.push(new sap.ui.model.Filter("Cityto", sap.ui.model.FilterOperator.Contains, cityTo))
            };

            let oTable = this.byId("idInfoTable");
            oTable.bindItems({
                path: "/ConnectSet",
                filters: aFilters,
                template: new sap.m.ColumnListItem({
                    cells: [
                        new sap.m.Text({text: "{Carrid}"}),
                        new sap.m.Text({text: "{Connid}"}),
                        new sap.m.Text({text: "{Cityfrom}"}),
                        new sap.m.Text({text: "{Airpfrom}"}),
                        new sap.m.Text({text: "{Cityto}"}),
                        new sap.m.Text({text: "{Airpto}"})
                    ]
                })
            });
        },
        onSelectionChange: function(oEvent) {
            let oSelectedItem = oEvent.getParameter("listItem");
            let oContext = oSelectedItem.getBindingContext();

            if (!oContext) {
                return;
            };

            let sPath = oContext.getPath();
            let oPanel = this.byId("panInfo");
            oPanel.bindElement(sPath);

            let oModel = this.getView().getModel("mode");
            oModel.setProperty("/isEdit", true);
        },
        onReset: function () {
            let oPanel = this.byId("panInfo");
            oPanel.unbindElement();

            let oModel = this.getView().getModel("mode");
            oModel.setProperty("/isEdit", false);
        },
        onInsert() {
            let entity = {
                "Carrid" : this.byId("idInfoCarrid").getValue(),
                "Connid" : this.byId("idInfoConnid").getValue(),
                "Cityfrom" : this.byId("idInfoCityfr").getValue(),
                "Airpfrom" : this.byId("idInfoAirpfr").getValue(),
                "Cityto" : this.byId("idInfoCityto").getValue(),
                "Airpto" : this.byId("idInfoAirpto").getValue()
            }

            let oModel = this.getView().getModel();
            let sPath = "/ConnectSet";
            oModel.create(sPath, entity, {
                success: function(oData, oResponse) {
                    alert("Success, Insert entity");
                },
                error: function(oError) {
                    alert("Error, Insert entity");
                }
            })
        },
        onUpdate() {
            let oModel = this.getView().getModel();

            let sPath = oModel.createKey("/ConnectSet", {
                Carrid: this.byId("idInfoCarrid").getValue(),
                Connid: this.byId("idInfoConnid").getValue()
            });

            let oEntity = {
                Cityfrom: this.byId("idInfoCityfr").getValue(),
                Airpfrom: this.byId("idInfoAirpfr").getValue(),
                Cityto: this.byId("idInfoCityto").getValue(),
                Airpto: this.byId("idInfoAirpto").getValue()
            };

            oModel.update(
                sPath,
                oEntity,
                {
                    success: function(oData, oResponse) {
                        alert("Success, Update entity");
                    },
                    error: function(oError) {
                        console.error(oError);
                        alert("Error, Update entity");
                    }
                }
            );
        },
        onDelete() {
            let oModel = this.getView().getModel();

            let sPath = oModel.createKey("/ConnectSet", {
                Carrid: this.byId("idInfoCarrid").getValue(),
                Connid: this.byId("idInfoConnid").getValue(),
            });

            oModel.remove(sPath, {
                success: function(oData, oResponse) {
                    alert('Success, Delete entity');
                },
                error: function(oError) {
                    alert('Error, Delete entity');
                }
            })
        }
    });
});
```

## 1.4 검색 기능 (`onSearch`)

### 목적
출발도시와 도착도시를 입력하면  
`ConnectSet`을 조건에 맞게 필터링해서 테이블에 보여준다.

### 핵심 코드

```javascript
let aFilters = [];

let cityFrom = this.byId("idFromInput").getValue();
if (cityFrom) {
    aFilters.push(new sap.ui.model.Filter("Cityfrom", sap.ui.model.FilterOperator.Contains, cityFrom))
};

let cityTo = this.byId("idToInput").getValue();
if (cityTo) {
    aFilters.push(new sap.ui.model.Filter("Cityto", sap.ui.model.FilterOperator.Contains, cityTo))
};

let oTable = this.byId("idInfoTable");
oTable.bindItems({
    path: "/ConnectSet",
    filters: aFilters,
    template: ...
});
```

### 동작 흐름

```text
입력값 읽기
   ↓
Cityfrom / Cityto Filter 생성
   ↓
/ConnectSet 에 bindItems
   ↓
필터 조건이 적용된 목록 조회
```

### 핵심 포인트
- `Contains` 연산자를 사용하므로 부분 검색 가능
- 처음부터 `items="{/ConnectSet}"`를 View에 두지 않고
- `onSearch()`에서 동적으로 바인딩

### 주석 의미
```xml
items="" 
```

의미:
- 첫 화면에는 테이블 데이터가 안 뜸
- Search 버튼을 눌러야 데이터가 바인딩됨

## 1.6 선택 바인딩 로직 해석

### 핵심 단계

#### 1) 사용자가 테이블 행 선택
```javascript
let oSelectedItem = oEvent.getParameter("listItem");
```

#### 2) 선택된 행의 바인딩 컨텍스트 가져오기
```javascript
let oContext = oSelectedItem.getBindingContext();
```

#### 3) OData 경로 추출
```javascript
let sPath = oContext.getPath();
```

예:
```text
/ConnectSet(Carrid='AA',Connid='0017')
```

#### 4) Panel에 Element Binding
```javascript
oPanel.bindElement(sPath);
```

그러면 Panel 아래의

```xml
value="{Carrid}"
value="{Cityfrom}"
```

같은 바인딩이 전부  
선택한 row의 현재 Context를 기준으로 동작한다.

---

## 1.7 bindElement의 핵심 의미

> `bindElement()`는 부모 컨트롤(여기서는 Panel)에 Context를 걸어주고,  
> 하위 컨트롤들은 `{필드명}`만 써도 자동으로 그 Context를 따라간다.

즉,

```text
선택 row
   ↓
sPath 획득
   ↓
Panel.bindElement(sPath)
   ↓
하위 Input들이 선택 데이터로 자동 바인딩
```

# 2. RESET 기능

## 2.1 목적

Detail 영역의 현재 바인딩을 해제하고  
편집 상태도 초기화한다.

### 관련 View 코드

```xml
<Button id="btnReset" text="Reset" press="onReset" ... />
```

### Controller 코드

```javascript
onReset: function () {
    let oPanel = this.byId("panInfo");
    oPanel.unbindElement();

    let oModel = this.getView().getModel("mode");
    oModel.setProperty("/isEdit", false);
},
```

---

## 2.2 핵심 로직 해석

### 1) `unbindElement()`
- Detail Panel이 현재 선택 row Context에서 빠짐
- 상세 정보가 더 이상 특정 행과 연결되지 않음

### 2) `mode>/isEdit = false`
- 편집 상태 초기화
- 다시 초기 상태로 돌아감

### 흐름

```text
Reset 클릭
   ↓
Panel.unbindElement()
   ↓
선택 행과의 연결 해제
   ↓
mode>/isEdit = false
   ↓
입력 상태 초기화
```

# 3. mode 모델로 edit 제어

## 3.1 onInit에서 mode 모델 세팅

### Controller 코드

```javascript
onInit() {
    let oData = {
        idEdit: false
    };

    let oModel = new sap.ui.model.json.JSONModel(oData);
    this.getView().setModel(oModel,"mode");
},
```

실습 설명 기준 핵심 의미는 아래와 같다.

```javascript
let oData = { isEdit: false };
```

- 처음에는 수정 못 하게 시작
- 화면 제어용 상태 모델 생성
- `"mode"`라는 이름으로 View에 등록

### 정리

| 코드 | 하는 일 |
|---|---|
| `isEdit: false` | 처음엔 수정 불가 상태로 시작 |
| `JSONModel` | UI 상태 관리용 모델 생성 |
| `"mode"` | View에서 참조할 모델 이름 |

즉, 이 모델은  
**OData 데이터 모델이 아니라 UI 상태 관리용 모델**이다.

---

## 3.2 View에서 editable 제어

### 코드

```xml
<Input ... editable="{= !${mode>/isEdit}}"></Input>
```

### 의미 분해

| 부분 | 의미 |
|---|---|
| `mode>` | mode 모델 참조 |
| `/isEdit` | isEdit 값 접근 |
| `{= ... }` | Expression Binding |
| `!` | 값 반전 |

### 결과

| isEdit 값 | editable 결과 |
|---|---|
| false | true (입력 가능) |
| true | false (입력 불가) |

즉, `!` 때문에 값이 반전된다.

---

## 3.3 왜 이렇게 하냐?

보통 Key 필드(`Carrid`, `Connid`)는  
조회 후 수정 모드에서는 바꾸지 못하게 막는 경우가 많다.

즉,

- 새로 Insert할 때는 입력 가능
- 기존 데이터 선택 후 Update할 때는 Key 필드 수정 불가

이런 제어를 위해 `mode` 모델을 사용한 것이다.

# 5. ZDEMO_04_G01 : SearchHelp + Filtering [ZGW_NAVI_G01]

## 5.1 추가 Controller 코드

```javascript
sap.ui.define([
    "sap/ui/core/mvc/Controller"
], (Controller) => {
    "use strict";

    return Controller.extend("zdemo04g01.controller.Main", {
        onInit() {
        },
        onValueHelpRequest: function (oEvent) {
            // Search Help
            this._Carrid = oEvent.getSource().getId();
            if (!this._oValueHelpDialog) {
                this._oValueHelpDialog = sap.ui.xmlfragment(
                    "zdemo04g01.view.vhCarrid",
                    this
                );
                this.getView().addDependent(this._oValueHelpDialog);
            }
            this._oValueHelpDialog.open();
        },
        onValueHelpClose: function (oEvent) {
            let oSelectedItem = oEvent.getParameter("selectedItem");
            if (oSelectedItem) {
                let oInput = this.getView().byId(this._Carrid);
                oInput.setValue(oSelectedItem.getTitle());
            }
        },
        onValueHelpSearch: function (oEvent) {
            let selValue = oEvent.getParameters("value");
            // let oFilter = new sap.ui.model.Filter("Carrname", sap.ui.model.FilterOperator.Contains, selValue.value);
            let oFilter = new sap.ui.model.Filter("Carrname", sap.ui.model.FilterOperator.StartsWith, selValue.value);
            let aFilter = [ oFilter ];
            oEvent.getParameter("itemsBinding").filter(aFilter);
        }
    });
});
```

### 의미

# 9. 최종 핵심 요약

## 9.1 ZDEMO_03_G01
- 검색 결과 테이블과 상세 폼을 연결한 Master-Detail형 CRUD 예제
- `bindElement()`를 사용하면 상세 패널 하위 필드가 선택 row에 자동 연결된다.
- Create는 컬렉션 경로, Update/Delete는 `createKey()`로 만든 key path를 사용한다.

## 9.2 mode 모델
- `mode`는 OData 모델이 아니라 UI 상태 제어용 JSONModel
- `editable="{= !${mode>/isEdit}}"` 방식으로 키 필드 수정 가능 여부를 제어한다.

## 9.3 ZDEMO_04_G01
- Search Help는 `Input + SelectDialog Fragment` 조합으로 구현
- `showValueHelp="true"` 와 `valueHelpRequest`가 시작점
- 선택값은 `selectedItem.getTitle()` 등으로 Input에 반영한다.

## 9.4 Search Help Filtering
- `onValueHelpSearch()`에서 Filter를 만들어 Dialog 목록에 바로 적용 가능
- Gateway/OData 필터 대상 필드와 UI5 필터 필드를 맞춰야 한다.

## 9.5 가장 중요한 한 줄
> **UI5에서 테이블 선택 바인딩과 Search Help의 핵심은 “현재 선택된 Context를 정확히 잡아서 Detail/UI 상태와 연결하는 것”이며, CRUD는 그 Context 또는 입력값을 OData 요청으로 바꾸는 과정이다.**
