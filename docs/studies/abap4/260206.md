```csharp
FORM SET_PROMO_RANGE.
  DATA LV_TODAY TYPE D.
  LV_TODAY = SY-DATUM.

  CASE GV_PROMO.
    WHEN ''.
      GV_DATE_FROM = LV_TODAY.
      GV_DATE_TO   = LV_TODAY + 28.
    WHEN '30'.
      GV_DATE_FROM = LV_TODAY.
      GV_DATE_TO   = LV_TODAY + 7.
    WHEN '40'.
      GV_DATE_FROM = LV_TODAY + 7.
      GV_DATE_TO   = LV_TODAY + 14.
    WHEN '50'.
      GV_DATE_FROM = LV_TODAY + 14.
      GV_DATE_TO   = LV_TODAY + 21.
    WHEN '60'.
      GV_DATE_FROM = LV_TODAY + 21.
      GV_DATE_TO   = LV_TODAY + 28.
    WHEN OTHERS.
      GV_PROMO     = ''.
      GV_DATE_FROM = LV_TODAY.
      GV_DATE_TO   = LV_TODAY + 28.
  ENDCASE.
ENDFORM.
```

```csharp
DATA: GO_CONT TYPE REF TO CL_GUI_CUSTOM_CONTAINER, "TOP에 추가 변수"

         SEATSOCC_B TYPE SFLIGHT-SEATSOCC_B,   "Business Occupied

         SEATSOCC_F TYPE SFLIGHT-SEATSOCC_F,   "First Occupied
```

```csharp
FORM SELECT_SFLIGHT. "조회 구문"
  CLEAR GT_OUT. 

  IF GV_CARRID IS INITIAL AND GV_CONNID IS INITIAL.

    SELECT CARRID, CONNID, FLDATE,
           PRICE, CURRENCY,
           SEATSMAX, SEATSOCC,
           SEATSMAX_B, SEATSOCC_B,
           SEATSMAX_F, SEATSOCC_F
      FROM SFLIGHT
      INTO TABLE @GT_OUT
      WHERE FLDATE BETWEEN @GV_DATE_FROM AND @GV_DATE_TO.

  ELSEIF GV_CARRID IS NOT INITIAL AND GV_CONNID IS INITIAL.

    SELECT CARRID, CONNID, FLDATE,
           PRICE, CURRENCY,
           SEATSMAX, SEATSOCC,
           SEATSMAX_B, SEATSOCC_B,
           SEATSMAX_F, SEATSOCC_F
      FROM SFLIGHT
      INTO TABLE @GT_OUT
      WHERE FLDATE BETWEEN @GV_DATE_FROM AND @GV_DATE_TO
        AND CARRID = @GV_CARRID.

  ELSEIF GV_CARRID IS INITIAL AND GV_CONNID IS NOT INITIAL.

    SELECT CARRID, CONNID, FLDATE,
           PRICE, CURRENCY,
           SEATSMAX, SEATSOCC,
           SEATSMAX_B, SEATSOCC_B,
           SEATSMAX_F, SEATSOCC_F
      FROM SFLIGHT
      INTO TABLE @GT_OUT
      WHERE FLDATE BETWEEN @GV_DATE_FROM AND @GV_DATE_TO
        AND CONNID = @GV_CONNID.

  ELSE.

    SELECT CARRID, CONNID, FLDATE,
           PRICE, CURRENCY,
           SEATSMAX, SEATSOCC,
           SEATSMAX_B, SEATSOCC_B,
           SEATSMAX_F, SEATSOCC_F
      FROM SFLIGHT
      INTO TABLE @GT_OUT
      WHERE FLDATE BETWEEN @GV_DATE_FROM AND @GV_DATE_TO
        AND CARRID = @GV_CARRID
        AND CONNID = @GV_CONNID.

  ENDIF.

  IF GT_OUT IS INITIAL.
    MESSAGE '조회 결과가 없습니다.' TYPE 'S'.
  ENDIF.

ENDFORM.
```

### Level 5  ALV 출력

```csharp
MODULE INIT_ALV OUTPUT. "스크린 100 PAI 모듈 추가"
  IF GO_CONT IS INITIAL.

    CREATE OBJECT GO_CONT
      EXPORTING
        CONTAINER_NAME = 'AREA'.

    CREATE OBJECT GO_ALV
      EXPORTING
        I_PARENT = GO_CONT.

    PERFORM SET_FIELDCAT_0100. "서브루틴 2개 생성"
    PERFORM SET_LAYOUT_0100.

    CALL METHOD GO_ALV->SET_TABLE_FOR_FIRST_DISPLAY
      EXPORTING
        IS_LAYOUT       = GS_LAYO
      CHANGING
        IT_OUTTAB       = GT_ALV
        IT_FIELDCATALOG = GT_FCAT.

  ENDIF.
ENDMODULE.
```

```csharp
FORM SET_FIELDCAT_0100.

  CLEAR GT_FCAT.

  PERFORM ADD_FCAT USING 'CARRID'     'ID'        'X'.
  PERFORM ADD_FCAT USING 'CONNID'     'No.'       'X'.
  PERFORM ADD_FCAT USING 'FLDATE'     'Flight Date' 'X'.

  PERFORM ADD_FCAT USING 'PRICE'      'Airfare'   ''.
  PERFORM ADD_FCAT USING 'CURRENCY'   'Curr.'     ''.

  PERFORM ADD_FCAT USING 'SEATSMAX'   'Capacity'  ''.
  PERFORM ADD_FCAT USING 'SEATSOCC'   'Occupied'  ''.

  PERFORM ADD_FCAT USING 'SEATSMAX_B' 'Capacity'  ''.
  PERFORM ADD_FCAT USING 'SEATSOCC_B' 'Occupied'  ''.

  PERFORM ADD_FCAT USING 'SEATSMAX_F' 'Capacity'  ''.
  PERFORM ADD_FCAT USING 'SEATSOCC_F' 'Occupied 1st' ''.

ENDFORM.
```

```csharp
FORM ADD_FCAT USING PV_FIELD PV_TEXT PV_KEY.
  CLEAR GS_FCAT.
  GS_FCAT-FIELDNAME = PV_FIELD.
  GS_FCAT-COLTEXT   = PV_TEXT.

  IF PV_KEY = 'X'.
    GS_FCAT-KEY = ABAP_TRUE.
  ENDIF.

  APPEND GS_FCAT TO GT_FCAT.
ENDFORM.
```

```csharp
FORM SET_LAYOUT_0100.
  CLEAR GS_LAYO.

  GS_LAYO-SEL_MODE = 'A'.
  GS_LAYO-GRID_TITLE = 'Promotion Flight List'.

ENDFORM.
```
