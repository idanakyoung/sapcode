```csharp
FORM GET_DATA_100 .
  CLEAR GT_ALV_100.
  REFRESH GT_ALV_100.

  SELECT
    A~AGENCYNUM,
    A~NAME,
    A~STREET,
    A~POSTCODE,
    A~CITY,
    A~COUNTRY,
    T~LANDX,
    A~TELEPHONE,
    A~URL,
    COUNT( B~BOOKID ) AS CNT_RES
    FROM STRAVELAG AS A
    LEFT JOIN SBOOK AS B
      ON A~AGENCYNUM = B~AGENCYNUM
     AND B~CANCELLED <> 'X'
     AND B~FLDATE BETWEEN @GS_SEL-FLDATE_LOW AND @GS_SEL-FLDATE_HIGH
    LEFT JOIN T005T AS T
      ON A~COUNTRY = T~LAND1
     AND T~SPRAS   = @SY-LANGU
    WHERE A~AGENCYNUM BETWEEN @GS_SEL-AGENCY_LOW AND @GS_SEL-AGENCY_HIGH
    GROUP BY
      A~AGENCYNUM, A~NAME, A~STREET, A~POSTCODE, A~CITY, A~COUNTRY,
      T~LANDX, A~TELEPHONE, A~URL
    INTO CORRESPONDING FIELDS OF TABLE @GT_ALV_100.

  GV_COUNT = LINES( GT_ALV_100 ).
  SET TITLEBAR 'T100' WITH GV_COUNT.
  "SET TITLEBAR 'T100' WITH LINES(GT_ALV_100).

  LOOP AT GT_ALV_100 INTO GS_ALV_100.

    "예약건수 0이면 라인 빨강.
    IF GS_ALV_100-CNT_RES = 0.
      GS_ALV_100-ROWCOLOR = 'C610'.  "예시: red 계열(시스템 테마 따라 다름)
    ENDIF.

    "예약건수 컬럼 노란색(셀 컬러).
    DATA(LS_SCOL) = VALUE LVC_S_SCOL(
                      FNAME = 'CNT_RES'

    APPEND LS_SCOL TO GS_ALV_100-CTAB.
    MODIFY GT_ALV_100 FROM GS_ALV_100.

  ENDLOOP.
ENDFORM.
```

```csharp
CLASS LCL_EVENT_HANDLER DEFINITION.
  PUBLIC SECTION.
```

```csharp
FORM GET_DATA_200 .

  CLEAR GT_ALV_200.
  REFRESH GT_ALV_200.

  SELECT
      B~CARRID,
      B~CONNID,
      B~FLDATE,
      B~BOOKID,
      B~CUSTOMID,
      C~NAME                AS NAME,
      B~CUSTTYPE,
      B~SMOKER,
      B~INVOICE,
      B~CLASS,
      B~LOCCURAM,
      B~LOCCURKEY,
      B~ORDER_DATE,
      P~DISTANCE,
      P~DISTID,
      Z~EARNMILES           AS EARNMILES
    FROM SBOOK AS B
    LEFT JOIN SCUSTOM AS C
      ON C~ID = B~CUSTOMID
    LEFT JOIN SPFLI AS P
      ON P~CARRID = B~CARRID
     AND P~CONNID = B~CONNID
    LEFT JOIN ZTC1003_01 AS Z
      ON Z~BOOKID = B~BOOKID 
    WHERE B~AGENCYNUM = @GV_AGENCYNUM
      AND B~CANCELLED <> 'X'
      AND B~FLDATE BETWEEN @GS_SEL-FLDATE_LOW AND @GS_SEL-FLDATE_HIGH
    INTO CORRESPONDING FIELDS OF TABLE @GT_ALV_200.
```

```csharp
FORM FILL_DOMAIN_TEXT_200 .
  DATA: LT_DOM TYPE TABLE OF DD07V,
        LS_DOM TYPE DD07V.

  " CUSTTYPE 도메인
  CALL FUNCTION 'GET_DOMAIN_VALUES'
    EXPORTING
      DOMNAME    = 'S_CUSTTYPE'
      TEXT       = 'X'
    TABLES
      VALUES_TAB = LT_DOM.

  LOOP AT GT_ALV_200 INTO GS_ALV_200.
    READ TABLE LT_DOM INTO LS_DOM WITH KEY DOMVALUE_L = GS_ALV_200-CUSTTYPE.
    IF SY-SUBRC = 0.
      GS_ALV_200-CUSTTYPE_T = LS_DOM-DDTEXT.
    ENDIF.
    MODIFY GT_ALV_200 FROM GS_ALV_200.
  ENDLOOP.

  " CLASS 도메인
  CLEAR LT_DOM.
  CALL FUNCTION 'GET_DOMAIN_VALUES'
    EXPORTING
      DOMNAME    = 'S_CLASS'
      TEXT       = 'X'
    TABLES
      VALUES_TAB = LT_DOM.

  LOOP AT GT_ALV_200 INTO GS_ALV_200.
    READ TABLE LT_DOM INTO LS_DOM WITH KEY DOMVALUE_L = GS_ALV_200-CLASS.
    IF SY-SUBRC = 0.
      GS_ALV_200-CLASS_T = LS_DOM-DDTEXT.
    ENDIF.
    MODIFY GT_ALV_200 FROM GS_ALV_200.
  ENDLOOP.
ENDFORM.
```

*PERFORM FILL_DOMAIN_TEXT_300.*

```csharp
FORM FILL_DOMAIN_TEXT_300 .
  DATA: LT_DOM TYPE TABLE OF DD07V,
        LS_DOM TYPE DD07V.

  " CLASS 텍스트 (S_CLASS)
  CLEAR LT_DOM.
  CALL FUNCTION 'GET_DOMAIN_VALUES'
    EXPORTING
      DOMNAME    = 'S_CLASS'
      TEXT       = 'X'
    TABLES
      VALUES_TAB = LT_DOM.

  LOOP AT GT_ALV_300 INTO GS_ALV_300.
    READ TABLE LT_DOM INTO LS_DOM
      WITH KEY DOMVALUE_L = GS_ALV_300-CLASS.
    IF SY-SUBRC = 0.
      GS_ALV_300-CLASS_T = LS_DOM-DDTEXT.
    ENDIF.
    MODIFY GT_ALV_300 FROM GS_ALV_300.
  ENDLOOP.

  " Customer Level 텍스트 (ZCUSTOMLV_01)
  CLEAR LT_DOM.
  CALL FUNCTION 'GET_DOMAIN_VALUES'
    EXPORTING
      DOMNAME    = 'ZCUSTOMLV_01'
      TEXT       = 'X'
    TABLES
      VALUES_TAB = LT_DOM.

  LOOP AT GT_ALV_300 INTO GS_ALV_300.
    READ TABLE LT_DOM INTO LS_DOM
      WITH KEY DOMVALUE_L = GS_ALV_300-CUSTOMLV.
    IF SY-SUBRC = 0.
      GS_ALV_300-CUSTOMLV_T = LS_DOM-DDTEXT.
    ENDIF.
    MODIFY GT_ALV_300 FROM GS_ALV_300.
  ENDLOOP.

ENDFORM.
```

```csharp
FORM PREPARE_EARN_210 USING PS_ROW TYPE TY_ALV_200.
  
  CLEAR GS_EARN_210.

  GS_EARN_210-CARRID   = PS_ROW-CARRID.
  GS_EARN_210-CONNID   = PS_ROW-CONNID.
  GS_EARN_210-FLDATE   = PS_ROW-FLDATE.
  GS_EARN_210-BOOKID   = PS_ROW-BOOKID.
  GS_EARN_210-CUSTOMID = PS_ROW-CUSTOMID.
  GS_EARN_210-CLASS    = PS_ROW-CLASS.
  GS_EARN_210-DISTANCE = PS_ROW-DISTANCE.
  GS_EARN_210-DISTID   = PS_ROW-DISTID.
  GS_EARN_210-EARNDATE = SY-DATUM.

  "Base Rate

  SELECT SINGLE STDRATE
    FROM ZTC1001_01
    WHERE CARRID = @GS_EARN_210-CARRID
      AND CLASS  = @GS_EARN_210-CLASS
    INTO @GS_EARN_210-STDRATE.

    IF SY-SUBRC <> 0 OR GS_EARN_210-STDRATE IS INITIAL.
      GS_EARN_210-STDRATE = 100.
    ENDIF.

  "Customer Level / Bonus Rate
```
