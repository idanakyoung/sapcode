```csharp
FORM GENERATE_LOTTO.

  DATA: LV_ROUND TYPE I,
        LT_NUMS  TYPE STANDARD TABLE OF I WITH EMPTY KEY,
        LV_NUM   TYPE I.

  PERFORM INIT_RAND.  " ← 여기서만 초기화

  CLEAR GT_LOTTO.

  DO GV_TRY TIMES.
    LV_ROUND = SY-INDEX.
    CLEAR GS_LOTTO.
    GS_LOTTO-ROUND = LV_ROUND.

    CLEAR LT_NUMS.

    WHILE LINES( LT_NUMS ) < 6.
      LV_NUM = GO_RAND->GET_NEXT( ).

      READ TABLE LT_NUMS WITH KEY TABLE_LINE = LV_NUM TRANSPORTING NO FIELDS.
      IF SY-SUBRC <> 0.
        APPEND LV_NUM TO LT_NUMS.
      ENDIF.
    ENDWHILE.

    SORT LT_NUMS.

    READ TABLE LT_NUMS INDEX 1 INTO GS_LOTTO-N1.
    READ TABLE LT_NUMS INDEX 2 INTO GS_LOTTO-N2.
    READ TABLE LT_NUMS INDEX 3 INTO GS_LOTTO-N3.
    READ TABLE LT_NUMS INDEX 4 INTO GS_LOTTO-N4.
    READ TABLE LT_NUMS INDEX 5 INTO GS_LOTTO-N5.
    READ TABLE LT_NUMS INDEX 6 INTO GS_LOTTO-N6.

    APPEND GS_LOTTO TO GT_LOTTO.
  ENDDO.

ENDFORM.
```

FORM GENERATE_WIN.

  DATA: LT_NUMS TYPE STANDARD TABLE OF I WITH EMPTY KEY,
        LV_NUM  TYPE I.

  PERFORM INIT_RAND.

  CLEAR: GT_WIN, GS_WIN.
  CLEAR LT_NUMS.

  "6개 번호 생성 (중복 X)
  WHILE LINES( LT_NUMS ) < 6.

    ENDIF.
  ENDWHILE.

  READ TABLE LT_NUMS INDEX 2 INTO GS_WIN-N2.
  READ TABLE LT_NUMS INDEX 3 INTO GS_WIN-N3.
  READ TABLE LT_NUMS INDEX 4 INTO GS_WIN-N4.
  READ TABLE LT_NUMS INDEX 5 INTO GS_WIN-N5.
  READ TABLE LT_NUMS INDEX 6 INTO GS_WIN-N6.

  "보너스 번호(6개와 중복 X)

    ENDIF.

```csharp
FORM CALC_RESULT.

  DATA: LT_WIN6  TYPE STANDARD TABLE OF I WITH EMPTY KEY,
        LV_BONUS TYPE I,
        LV_HIT   TYPE I,
        LV_BHIT  TYPE ABAP_BOOL.

  DATA LS_SCOL TYPE LVC_S_SCOL.

  "방어
  IF GT_LOTTO IS INITIAL.
    CLEAR GT_RESULT.
    MESSAGE '먼저 번호 생성(번호 생성 버튼)부터 하세요.' TYPE 'S'.
    RETURN.
  ENDIF.

  READ TABLE GT_WIN INDEX 1 INTO GS_WIN.
  IF SY-SUBRC <> 0.
    CLEAR GT_RESULT.
    MESSAGE '당첨번호가 없습니다. 실행을 다시 하세요.' TYPE 'S'.
    RETURN.
  ENDIF.

  "당첨 6개 + 보너스
  CLEAR LT_WIN6.
  APPEND GS_WIN-N1 TO LT_WIN6.
  APPEND GS_WIN-N2 TO LT_WIN6.
  APPEND GS_WIN-N3 TO LT_WIN6.
  APPEND GS_WIN-N4 TO LT_WIN6.
  APPEND GS_WIN-N5 TO LT_WIN6.
  APPEND GS_WIN-N6 TO LT_WIN6.
  SORT LT_WIN6.
  LV_BONUS = GS_WIN-BONUS.

  CLEAR GT_RESULT.

  LOOP AT GT_LOTTO INTO GS_LOTTO.

    LV_HIT  = 0.
    LV_BHIT = ABAP_FALSE.

    "맞춘 개수
    PERFORM HIT_CHECK USING GS_LOTTO-N1 LT_WIN6 CHANGING LV_HIT.
    PERFORM HIT_CHECK USING GS_LOTTO-N2 LT_WIN6 CHANGING LV_HIT.
    PERFORM HIT_CHECK USING GS_LOTTO-N3 LT_WIN6 CHANGING LV_HIT.
    PERFORM HIT_CHECK USING GS_LOTTO-N4 LT_WIN6 CHANGING LV_HIT.
    PERFORM HIT_CHECK USING GS_LOTTO-N5 LT_WIN6 CHANGING LV_HIT.
    PERFORM HIT_CHECK USING GS_LOTTO-N6 LT_WIN6 CHANGING LV_HIT.

    "보너스 포함 여부(내 번호 6개 중 보너스가 있는지)
    IF GS_LOTTO-N1 = LV_BONUS OR GS_LOTTO-N2 = LV_BONUS OR GS_LOTTO-N3 = LV_BONUS OR
       GS_LOTTO-N4 = LV_BONUS OR GS_LOTTO-N5 = LV_BONUS OR GS_LOTTO-N6 = LV_BONUS.
      LV_BHIT = ABAP_TRUE.
    ENDIF.

    "결과 라인 기본값
    CLEAR GS_RESULT.
    GS_RESULT-ROUND = GS_LOTTO-ROUND.
    GS_RESULT-N1    = GS_LOTTO-N1.
    GS_RESULT-N2    = GS_LOTTO-N2.
    GS_RESULT-N3    = GS_LOTTO-N3.
    GS_RESULT-N4    = GS_LOTTO-N4.
    GS_RESULT-N5    = GS_LOTTO-N5.
    GS_RESULT-N6    = GS_LOTTO-N6.
    CLEAR GS_RESULT-CELLCOLOR.

    "등수 결정
    GS_RESULT-GRADE = 0.
    CASE LV_HIT.
      WHEN 6.
        GS_RESULT-GRADE = 1.
      WHEN 5.
        IF LV_BHIT = ABAP_TRUE.
          GS_RESULT-GRADE = 2.
        ELSE.
          GS_RESULT-GRADE = 3.
        ENDIF.
      WHEN 4.
        GS_RESULT-GRADE = 4.
      WHEN 3.
        GS_RESULT-GRADE = 5.
      WHEN OTHERS.
        GS_RESULT-GRADE = 0.
    ENDCASE.

    "결과 텍스트(0이면 꽝)
    GS_RESULT-RESULT_TXT =
      COND STRING(
        WHEN GS_RESULT-GRADE = 0 THEN '꽝'
        ELSE |{ GS_RESULT-GRADE }등| ).

    "-----------------------------------------
    " (1) 당첨번호와 일치하는 셀 = 초록색
    " (보너스 일치까지 표시하고 싶으면 아래 보너스 색도 사용 가능)
    "-----------------------------------------
    PERFORM COLOR_IF_HIT USING 'N1' GS_RESULT-N1 LT_WIN6 LV_BONUS CHANGING GS_RESULT-CELLCOLOR.
    PERFORM COLOR_IF_HIT USING 'N2' GS_RESULT-N2 LT_WIN6 LV_BONUS CHANGING GS_RESULT-CELLCOLOR.
    PERFORM COLOR_IF_HIT USING 'N3' GS_RESULT-N3 LT_WIN6 LV_BONUS CHANGING GS_RESULT-CELLCOLOR.
    PERFORM COLOR_IF_HIT USING 'N4' GS_RESULT-N4 LT_WIN6 LV_BONUS CHANGING GS_RESULT-CELLCOLOR.
    PERFORM COLOR_IF_HIT USING 'N5' GS_RESULT-N5 LT_WIN6 LV_BONUS CHANGING GS_RESULT-CELLCOLOR.
    PERFORM COLOR_IF_HIT USING 'N6' GS_RESULT-N6 LT_WIN6 LV_BONUS CHANGING GS_RESULT-CELLCOLOR.

    "-----------------------------------------
    " (2) 결과가 1~5등이면 결과 셀 = 파란색
    "-----------------------------------------
    IF GS_RESULT-GRADE BETWEEN 1 AND 5.
      CLEAR LS_SCOL.
      LS_SCOL-FNAME = 'RESULT_TXT'.
      LS_SCOL-COLOR-COL = 1. "보통 1=Blue (시스템에 따라 다르면 4/5로 조정)
      LS_SCOL-COLOR-INT = 0.
      LS_SCOL-COLOR-INV = 0.
      APPEND LS_SCOL TO GS_RESULT-CELLCOLOR.
    ENDIF.

    APPEND GS_RESULT TO GT_RESULT.

  ENDLOOP.

ENDFORM.
```

FORM calc_result.

  DATA: lt_win6  TYPE STANDARD TABLE OF i WITH EMPTY KEY,

  IF sy-subrc <> 0.
    CLEAR gt_result.
    MESSAGE '당첨번호가 없습니다. 실행을 다시 하세요.' TYPE 'S'.
    RETURN.
  ENDIF.

  "당첨 6개 + 보너스
  CLEAR lt_win6.
  APPEND gs_win-n1 TO lt_win6.
  APPEND gs_win-n2 TO lt_win6.
  APPEND gs_win-n3 TO lt_win6.
  APPEND gs_win-n4 TO lt_win6.
  APPEND gs_win-n5 TO lt_win6.
  APPEND gs_win-n6 TO lt_win6.
  SORT lt_win6.
  lv_bonus = gs_win-bonus.

  CLEAR gt_result.

  LOOP AT gt_lotto INTO gs_lotto.

    lv_hit  = 0.
    lv_bhit = abap_false.

    "맞춘 개수
    PERFORM hit_check USING gs_lotto-n1 lt_win6 CHANGING lv_hit.
    PERFORM hit_check USING gs_lotto-n2 lt_win6 CHANGING lv_hit.
    PERFORM hit_check USING gs_lotto-n3 lt_win6 CHANGING lv_hit.
    PERFORM hit_check USING gs_lotto-n4 lt_win6 CHANGING lv_hit.
    PERFORM hit_check USING gs_lotto-n5 lt_win6 CHANGING lv_hit.
    PERFORM hit_check USING gs_lotto-n6 lt_win6 CHANGING lv_hit.

    "보너스 포함 여부(내 번호 6개 중 보너스가 있는지)
    IF gs_lotto-n1 = lv_bonus OR gs_lotto-n2 = lv_bonus OR gs_lotto-n3 = lv_bonus OR
       gs_lotto-n4 = lv_bonus OR gs_lotto-n5 = lv_bonus OR gs_lotto-n6 = lv_bonus.
      lv_bhit = abap_true.
    ENDIF.

    "결과 라인 기본값
    CLEAR gs_result.
    gs_result-round = gs_lotto-round.
    gs_result-n1    = gs_lotto-n1.
    gs_result-n2    = gs_lotto-n2.
    gs_result-n3    = gs_lotto-n3.
    gs_result-n4    = gs_lotto-n4.
    gs_result-n5    = gs_lotto-n5.
    gs_result-n6    = gs_lotto-n6.
    CLEAR gs_result-cellcolor.

    "등수 결정
    gs_result-grade = 0.
    CASE lv_hit.
      WHEN 6.
        gs_result-grade = 1.
      WHEN 5.
        IF lv_bhit = abap_true.
          gs_result-grade = 2.
        ELSE.
          gs_result-grade = 3.
        ENDIF.
      WHEN 4.
        gs_result-grade = 4.
      WHEN 3.
        gs_result-grade = 5.
      WHEN OTHERS.
        gs_result-grade = 0.
    ENDCASE.

    "결과 텍스트(0이면 꽝)
    gs_result-result_txt =
      COND string(
        WHEN gs_result-grade = 0 THEN '꽝'
        ELSE |{ gs_result-grade }등| ).

    "-----------------------------------------
    " (1) 당첨번호와 일치하는 셀 = 초록색
    " (보너스 일치까지 표시하고 싶으면 아래 보너스 색도 사용 가능)
    "-----------------------------------------
    PERFORM color_if_hit USING 'N1' gs_result-n1 lt_win6 lv_bonus CHANGING gs_result-cellcolor.
    PERFORM color_if_hit USING 'N2' gs_result-n2 lt_win6 lv_bonus CHANGING gs_result-cellcolor.
    PERFORM color_if_hit USING 'N3' gs_result-n3 lt_win6 lv_bonus CHANGING gs_result-cellcolor.
    PERFORM color_if_hit USING 'N4' gs_result-n4 lt_win6 lv_bonus CHANGING gs_result-cellcolor.
    PERFORM color_if_hit USING 'N5' gs_result-n5 lt_win6 lv_bonus CHANGING gs_result-cellcolor.
    PERFORM color_if_hit USING 'N6' gs_result-n6 lt_win6 lv_bonus CHANGING gs_result-cellcolor.

    "-----------------------------------------
    " (2) 결과가 1~5등이면 결과 셀 = 파란색
    "-----------------------------------------
    IF gs_result-grade BETWEEN 1 AND 5.
      CLEAR ls_scol.
      ls_scol-fname = 'RESULT_TXT'.
      ls_scol-color-col = 1. "보통 1=Blue (시스템에 따라 다르면 4/5로 조정)
      ls_scol-color-int = 0.
      ls_scol-color-inv = 0.
      APPEND ls_scol TO gs_result-cellcolor.
    ENDIF.

    APPEND gs_result TO gt_result.

  ENDLOOP.

ENDFORM.

FORM color_if_hit
  USING    pv_field TYPE lvc_fname
           pv_num   TYPE i
           pt_win6  TYPE STANDARD TABLE
           pv_bonus TYPE i
  CHANGING pt_scol  TYPE lvc_t_scol.

  DATA ls_scol TYPE lvc_s_scol.

  "당첨 6개 일치 -> 초록
  READ TABLE pt_win6 WITH KEY table_line = pv_num TRANSPORTING NO FIELDS.
  IF sy-subrc = 0.
    CLEAR ls_scol.
    ls_scol-fname = pv_field.
    ls_scol-color-col = 5. "보통 5=Green
    ls_scol-color-int = 0.
    ls_scol-color-inv = 0.
    APPEND ls_scol TO pt_scol.
    RETURN.
  ENDIF.

  "보너스까지 색 주고 싶으면(캡처에서 보너스는 다른색 느낌)
  IF pv_num = pv_bonus.
    CLEAR ls_scol.
    ls_scol-fname = pv_field.
    ls_scol-color-col = 3. "보통 3=Yellow
    ls_scol-color-int = 0.
    ls_scol-color-inv = 0.
    APPEND ls_scol TO pt_scol.
  ENDIF.

ENDFORM.
